\chapter{Notation Glossary}
\label{app:notation}

\section{Constants and Globally Defined Variables}
\begin{description}
    \item[$A_i$] Matrix in the family $\mathcal{A}$.
    \item[$d$] Dimension of each matrix $A_i \in \R^{d \times d}$.
    \item[$n$] Number of matrices in $\mathcal{A}$.
    \item[$\mathcal{A} = \{A_1,\dots,A_n\}$] Base set of matrices.
    \item[$\tilde{\mathcal{A}} = \{\tilde{A}_1,\dots,\tilde{A}_n\}$] Scaled matrix set after preprocessing.
    \item[$m$] Number of generators.
    \item[$\mathcal{G} = \{G_1,\dots,G_m\}$] Set of generators.
    \item[$k$] Length of a product.
    \item[$\alpha$] Scalar.
    \item[$\lambda$] Eigenvalue.
    \item[$x$] Vector.
    \item[$\rho$] Spectral radius.
    \item[$\sigma$] Singular value.
    \item[$\JSR$] Joint spectral radius operator.
\end{description}

\section{Products and Multi-Indices}
\begin{description}
    \item[$\mathcal{I} = \{1,\dots,n\}$] Positive matrix-index set.
    \item[$\mathcal{I}^k$] Set of positive multi-indices of length $k$.
    \item[$i$] Positive index.
    \item[$I$] Positive multi-index of length $k$, i.e. $I = [i_1,\dots,i_k] \in \mathcal{I}^k$.
    \item[$A_I = A_{[i_1,\dots,i_k]} = A_{i_k}\cdots A_{i_1}$] Product encoded by $I$.
    \item[$\mathcal{J} = \{-m,\dots,-1,1,\dots,n\}$] Extended index set including generator indices.
    \item[$\mathcal{J}^k$] Set of extended multi-indices of length $k$.
    \item[$j$] Extended index.
    \item[$\mathcal{A}_{\mathcal{G}}$] Family of primitive product sets:
    \[
        \mathcal{A}_{\mathcal{G}} :=
        \left\{
            \{G_m^r : r \in \N\},\dots,\{G_1^r : r \in \N\},\{A_1\},\dots,\{A_n\}
        \right\}.
    \]
    \item[$A_j \in \mathcal{A}_{\mathcal{G}}$] Primitive set encoded by index $j$.
    \item[$J$] Extended multi-index of length $k$, i.e. $J = [j_1,\dots,j_k] \in \mathcal{J}^k$.
    \item[$A_J = A_{[j_1,\dots,j_k]}$] Product set encoded by $J$.
    \item[${[J_1, j_3, J_4, [j_1, J_2]]}$] Expressions are interpreted in flattened form.
\end{description}

\section{Norms and JSR}
\begin{description}
    \item[$\|\cdot\|_{\mathrm{co}_{\mathrm{s}}(V)} = \|\cdot\|_V$] Polytope norm notation used for matrices and vectors.
    \item[$\JSR(\mathcal{A}) := \lim\limits_{k \to \infty} \max\limits_{P \in \mathcal{A}^k}\|P\|^{1/k}$] Norm-based definition of JSR.
    \item[$\JSR(\mathcal{A}) = \limsup\limits_{k \to \infty} \max\limits_{P \in \mathcal{A}^k}\rho(P)^{1/k}$] Berger--Wang characterization.
\end{description}

\section{Algorithmic Notation}
\begin{description}
    \item[Invariant-polytope algorithm] Baseline algorithm from Chapter~\ref{ch:invariantPolytope}.
    \item[Finite-tree algorithm] Tree-based algorithm from Chapter~\ref{ch:finiteTree}.
    \item[Invariant-tree algorithm] Combined algorithm from Chapter~\ref{ch:hybridApproach}.
    \item[$M$] Number of vertices in a finite vertex set representation $V=\{v_1,\dots,v_M\}$.
    \item[$V_k$] Set of vertices at step $k$.
    \item[$R_k$] Set of vertices remaining to test at step $k$.
    \item[$V$] Union of vertex sets over all steps.
    \item[$\xleftarrow{\cup}$] Appending by union in pseudocode.
\end{description}

\section{Tree Terminology}
\begin{description}
    \item[$T$] a tree is a directed graph.
    \item[Root] Node without a predecessor.
    \item[Child and parent] If a node $t \in T$ has predecessor $p \in T$, then $t$ is a child of $p$ and $p$ is the parent of $t$.
    \item[Leaf] Node without children.
    \item[Internal node] Node with at least one child.
    \item[$T_{\mathcal{A}}$] Infinite product tree generated by $\mathcal{A}$.
    \item[$\AG$-tree] Set-valued tree with root $\{I_d\}$ and internal nodes having either $n$ positive children or at least one generator child.
    \item[Covered or uncovered node] A node is covered if its set is a subset of an ancestor node; otherwise it is uncovered.
    \item[$\mathcal{L}(T)$] Leafage of $T$, i.e. all uncovered leaves.
    \item[$1$-bounded] All matrix sets encoded by $\mathcal{L}(T)$ are bounded by $1$ in a chosen matrix norm.
    \item[$v$-$1$-bounded] All images of matrices encoded by $\mathcal{L}(T)$ applied to a fixed vector $v$ are bounded by $1$ in a chosen vector norm.
    \item[PL decomposition] Prefix-leaf decomposition.
    \item[$T_{\min\AG}$] Minimal $\AG$-tree, i.e. the smallest such tree containing a covered node.
\end{description}
