\chapter{Finite-Tree Algorithm}
\label{ch:finiteTree}

This chapter summarizes the finite-tree approach from \citep{MoellerReif2014TreeBasedJSR}. Its role is to provide the first tree-based termination argument in the thesis and to bridge the invariant-polytope method from Chapter~\ref{ch:invariantPolytope} to the hybrid method in Chapter~\ref{ch:hybridApproach}.

\section{Overview}

\subsection{Motivating Ideas}
The finite-tree algorithm bounds arbitrary products $P \in \mathcal{A}^k$ by decomposing them into factors that are $1$-bounded with respect to a chosen norm and one residual factor that grows at most polynomially.

\subsection{Problem Formulation}
After preprocessing from Section~\ref{sec:introPreprocessing}, we obtain a rescaled family $\tilde{\mathcal{A}}$. From the chosen candidate product and the three-member inequality \eqref{eq:three_member}, we already know $\JSR(\tilde{\mathcal{A}}) \ge 1$. Hence it remains to prove $\JSR(\tilde{\mathcal{A}}) \le 1$.
To simplify notation, we write $\mathcal{A}$ for this rescaled family throughout the remainder of the chapter.

\subsection{Core Idea and Notation}
The finite-tree argument is the first generalization of the invariant-polytope strategy from Chapter~\ref{ch:invariantPolytope}. Instead of requiring one-step invariance of a balanced polytope, it organizes products in a finite tree and proves that every long product can be decomposed into leaf factors bounded by $1$ and one remaining factor with only polynomial growth.

We begin with the rooted product tree $T_{\mathcal{A}}$ whose nodes are all positive multi-indices. For the empty index, we set
\[
A_{[]} = I_d,
\]
and this node is the root.

In general, a direct search in $T_{\mathcal{A}}$ is not feasible. Following \citep{MoellerReif2014TreeBasedJSR}, one therefore introduces generators and set-valued nodes.
We fix a generator set
\[
\mathcal{G} = \{G_1,\ldots,G_m\},
\]
where each generator satisfies
\[
G_r \in \mathcal{A}^{k_r}
\quad \text{and} \quad
\rho(G_r)^{1/k_r} \le 1.
\]

\begin{definition}\citep{MoellerReif2014TreeBasedJSR}
    \label{def:extendedMultiIndexSet}
    Set
    \[
    \mathcal{J} := \{-m,\ldots,-1,1,\ldots,n\}
    \]
    and define
    \[
    A_{-r} := G_r
    \quad \text{for } r = 1,\ldots,m.
    \]
    For a multi-index $J = [j_1,\ldots,j_\ell] \in \mathcal{J}^{\ell}$, the associated set of matrix products is
    \[
    A_J :=
    \left\{
      A_{j_{\ell}}^{e_{\ell}} \cdots A_{j_1}^{e_1}
      :
      \begin{array}{l}
      e_s = 1 \text{ if } j_s > 0,\\
      e_s \in \mathbb{N}_0 \text{ if } j_s < 0
      \end{array}
    \right\}.
    \]
    Thus positive entries represent single matrices from $\mathcal{A}$, while negative entries represent arbitrary powers of generators.
\end{definition}

\begin{definition}\citep{MoellerReif2014TreeBasedJSR}
    \label{def:agTree}
    A finite rooted tree $T$ is called an \emph{$\AG$-tree} if its nodes are multi-indices from $\bigcup_{\ell \in \mathbb{N}_0} \mathcal{J}^{\ell}$, the root is the empty index $[]$, and every node is one of the following:
    \begin{itemize}
        \item a leaf,
        \item a parent of exactly $n$ positive children,
        \item or a parent with at least one generator child.
    \end{itemize}
\end{definition}

\begin{definition}\citep{MoellerReif2014TreeBasedJSR}
    \label{def:coveredNode}
    A node of an $\AG$-tree is called \emph{covered} if its associated product set is contained in the product set of one of its ancestors. Otherwise, it is called \emph{uncovered}. The set of uncovered leaves is denoted by
    \[
    \mathcal{L}(T) := \{K \in T : K \text{ is an uncovered leaf}\}
    \]
    and is called the \emph{leafage} of $T$.
\end{definition}

\begin{definition}\citep{MoellerReif2014TreeBasedJSR}
\label{def:oneBoundedTree}
An $\AG$-tree $T$ is called \emph{$1$-bounded} with respect to a matrix norm $\lVert \cdot \rVert$ if
\[
\sup\{\lVert P \rVert : L \in \mathcal{L}(T),\ P \in A_L\} \le 1.
\]
If
\[
\sup\{\lVert P \rVert : L \in \mathcal{L}(T),\ P \in A_L\} < 1,
\]
then $T$ is called \emph{strictly $1$-bounded}.
\end{definition}

\section{Algorithmic Formulation}
\label{sec:finiteAlgorithmStructure}

\subsection{Algorithm Description}
The finite-tree algorithm performs a non-exhaustive search in the space of $\AG$-trees. It starts from the positive children of the root.
We denote by $Q_i$ the set of active nodes that still have to be tested in iteration $i$.
For each active node $J \in Q_i$, the algorithm checks whether the associated product set $A_J$ is already $1$-bounded.
If not, the node is expanded according to a child-selection rule, and the new children are placed into the next active set $Q_{i+1}$.
If $Q_i = \emptyset$, then the current tree is a finite certificate candidate.

\subsection{Pseudocode}
\begin{algorithm}[ht]
    \caption{Finite-tree algorithm}
    \label{alg:finiteTree}
    \begin{algorithmic}
        \State \Input{$\mathcal{A}$ (rescaled family of matrices), $\mathcal{G}$ (set of generators)}
        \State $T_0 \gets \{[], [1], \ldots, [n]\}$
        \State $Q_0 \gets \{[1], \ldots, [n]\}$
        \State $i \gets 0$
        \While{$Q_i \ne \emptyset$}
            \State $T_{i+1} \gets T_i$
            \State $Q_{i+1} \gets \emptyset$
            \For{$J \in Q_i$}
                \If{$\sup\{\lVert P \rVert : P \in A_J\} > 1$}
                    \State $\mathcal{C} \gets \textproc{selectChildren}(J,\mathcal{A},\mathcal{G})$
                    \State $T_{i+1} \xleftarrow{\cup} \{[J,j] : j \in \mathcal{C}\}$
                    \State $Q_{i+1} \xleftarrow{\cup} \{[J,j] : j \in \mathcal{C}\}$
                \EndIf
            \EndFor
            \State $i \gets i + 1$
        \EndWhile
        \State \Return $T_i$
    \end{algorithmic}
\end{algorithm}

\section{Termination Theory}

\subsection{Proof Strategy}
The finite-tree argument is the tree-based analogue of the invariant-polytope argument from Chapter~\ref{ch:invariantPolytope}. Instead of proving one-step invariance of a set, it bounds arbitrary products by repeated prefix-leaf decomposition. The proof has two steps. First, one proves a polynomial bound for products encoded by nodes of a fixed finite $\AG$-tree. Second, one decomposes an arbitrary product into leaf factors bounded by $1$ and one residual node factor, then passes to the $k$-th root limit in the definition of $\JSR$.

\subsection{Auxiliary Results}
\begin{lemma}
    \label{lem:finitePolynomialBound}
    For every finite $\AG$-tree $T$ there exists a monotone polynomial $p$ such that for every node $K \in T$ and every product $P \in \mathcal{A}^k$ encoded by $K$, one has
    \[
    \lVert P \rVert \le p(k).
    \]
\end{lemma}

\begin{proof}
    Let $K = [j_1,\ldots,j_{\ell}]$ be a fixed node of $T$.
    For each position $r = 1,\ldots,\ell$, define a monotone polynomial $q_r$ as follows.
    If $j_r > 0$, then the corresponding factor is always the single matrix $A_{j_r}$, so we set
    \[
    q_r(t) := \lVert A_{j_r} \rVert
    \quad \text{for all } t \in \mathbb{N}_0.
    \]
    If $j_r < 0$, then $A_{j_r} = G_{-j_r}$ is a generator with spectral radius at most $1$.
    Hence the powers of $A_{j_r}$ grow at most polynomially by the Jordan-normal-form argument, so there exists a monotone polynomial $q_r$ such that
    \[
    \lVert A_{j_r}^{e} \rVert \le q_r(e)
    \quad \text{for all } e \in \mathbb{N}_0.
    \]

    Set
    \[
    p_K := q_{\ell} \cdots q_1
    \]
    and
    \[
    p := \sum_{K \in T} p_K.
    \]
    Since $T$ is finite, the function $p$ is a monotone polynomial.

    Now let
    \[
    P = A_{j_{\ell}}^{e_{\ell}} \cdots A_{j_1}^{e_1} \in \mathcal{A}^k
    \]
    be encoded by the node $K$.
    By submultiplicativity,
    \[
    \lVert P \rVert
    \le \lVert A_{j_{\ell}}^{e_{\ell}} \rVert \cdots \lVert A_{j_1}^{e_1} \rVert
    \le q_{\ell}(e_{\ell}) \cdots q_1(e_1).
    \]
    Every exponent satisfies $e_r \le k$, because the encoded product has total length $k$.
    Using the monotonicity of the polynomials $q_r$, we obtain
    \[
    \lVert P \rVert
    \le q_{\ell}(k) \cdots q_1(k)
    = p_K(k)
    \le p(k).
    \]
    Since the node $K$ and the product $P$ were arbitrary, the lemma follows.
\end{proof}

\begin{lemma}\citep[cf. Lemma~4.1 and the proof of Theorem~3.3]{MoellerReif2014TreeBasedJSR}
    \label{lem:agTreePLDecomposition}
    Let $T$ be a finite $\AG$-tree and let $I \in \mathcal{I}^k$.
    Then there exist unique multi-indices $L$ and $P$, each either positive or equal to $[]$, such that
    \[
    I = [L,P]
    \quad \text{and hence} \quad
    A_I = A_P A_L.
    \]
    Moreover, either $P = []$ and $A_I$ is encoded by a node of $T$, or $A_L$ is encoded by a leaf in $\mathcal{L}(T)$ and $|P| < k$.
\end{lemma}

\begin{proof}
    Let
    \[
    \mathcal{I}_{\ast}(T)
    :=
    \left\{
      J \in \bigcup_{r \in \mathbb{N}_0} \mathcal{I}^r
      :
      A_J \in \bigcup_{K \in T} A_K
    \right\}
    \]
    denote the set of positive multi-indices encoded by nodes of $T$.
    Choose the longest prefix $L$ of $I$ that belongs to $\mathcal{I}_{\ast}(T)$, and let $P$ be the complementary suffix, so that $I = [L,P]$.
    The maximality of the prefix makes this choice unique.

    If $P = []$, then the whole product $A_I$ is encoded by a node of $T$, and the first case holds.
    Assume therefore that $P \ne []$.
    Let $K$ be maximal among the nodes of $T$ whose encoded product set contains $A_L$.
    By maximality of the prefix $L$, the node $K$ cannot have a positive child continuing the word $I$.
    It also cannot have a generator child, because every generator child contains the power $0$ and would therefore still encode $A_L$ at a deeper node.
    Finally, the argument in \citep[proof of Lemma~4.1]{MoellerReif2014TreeBasedJSR} excludes the possibility that $K$ is covered.
    Hence $K$ is an uncovered leaf, that is, $K \in \mathcal{L}(T)$.
    Therefore $A_L$ is encoded by a leaf of $T$.
    Since $P \ne []$, one also has $|P| < k$.
\end{proof}

\subsection{Main Result}
\begin{theorem}\citep[Theorem~3.3]{MoellerReif2014TreeBasedJSR}
\label{thm:finiteTreeTermination}
    If the rescaled family $\mathcal{A}$ admits a finite $1$-bounded $\AG$-tree, then
    \[
    \JSR(\mathcal{A}) \le 1.
    \]
\end{theorem}

\begin{proof}
    Take an arbitrary positive multi-index $I \in \mathcal{I}^k$.
    Repeatedly apply Lemma~\ref{lem:agTreePLDecomposition} to the current remainder until the remaining factor is encoded by a node of $T$.
    This yields multi-indices $I_1,\ldots,I_r$ such that
    \[
    A_I = A_{I_r} \cdots A_{I_1},
    \]
    where each factor $A_{I_j}$ with $j = 1,\ldots,r-1$ is encoded by a leaf in $\mathcal{L}(T)$ and the last factor $A_{I_r}$ is encoded by a node of $T$.

    By Definition~\ref{def:oneBoundedTree}, the leaf factors satisfy
    \[
    \lVert A_{I_j} \rVert \le 1
    \quad \text{for } j = 1,\ldots,r-1.
    \]
    By Lemma~\ref{lem:finitePolynomialBound}, there exists a monotone polynomial $p$ such that
    \[
    \lVert A_{I_r} \rVert \le p(k).
    \]
    Therefore,
    \[
    \lVert A_I \rVert
    = \lVert A_{I_r} \cdots A_{I_1} \rVert
    \le \lVert A_{I_r} \rVert \cdots \lVert A_{I_1} \rVert
    \le p(k).
    \]

    Since the multi-index $I$ was arbitrary,
    \[
    \JSR(\mathcal{A})
    = \lim_{k \to \infty} \max_{P \in \mathcal{A}^k} \lVert P \rVert^{1/k}
    \le \lim_{k \to \infty} p(k)^{1/k}
    = 1.
    \]
\end{proof}

\begin{remark}
    Combined with the preprocessing step from Section~\ref{sec:introPreprocessing}, Theorem~\ref{thm:finiteTreeTermination} certifies that the chosen candidate scaling is correct. In this sense, the finite-tree method plays the same role for tree-based certificates that Theorem~\ref{thm:invariantPolytopeTermination} plays for invariant polytopes.
\end{remark}

\subsection{Termination Conditions}
Theorem~\ref{thm:finiteTreeTermination} gives the central sufficient condition: a finite $1$-bounded $\AG$-tree implies the desired upper bound for the rescaled family.
For actual finite termination of the search procedure, the child-selection rule must lead to such a tree after finitely many expansions.

\section{Implementation Details}

\subsection{Child-Selection Policy}
The function $\textproc{selectChildren}$ decides whether a violating node is expanded by all positive children or by a suitable collection of generator children. This choice is part of the practical search strategy and follows the rules discussed in \citep{MoellerReif2014TreeBasedJSR}.

\section{Remarks}

\begin{remark}\citep[Theorem~3.5]{MoellerReif2014TreeBasedJSR}
    If the rescaled family $\mathcal{A}$ is irreducible and has a spectral gap at $1$ in the sense of Definition~\ref{def:spectralGap}, then, with a suitable choice of norm, a finite tree of the type required in Theorem~\ref{thm:finiteTreeTermination} exists. Thus the finite-tree method is guaranteed to succeed in that setting.
\end{remark}

\begin{remark}\citep{MoellerReif2014TreeBasedJSR}
    Generators are essential for the strength of the method. They make it possible to treat infinite families of products through finite tree nodes and can substantially reduce the depth of the searched tree. This is one of the key differences between the finite-tree and invariant-polytope approaches.
\end{remark}
