\chapter{Invariant-Polytope Algorithm}
\label{ch:invariantPolytope}

This chapter summarizes the invariant-polytope approach from \citep{GuglielmiProtasov2013ExactComputationJSR} and related follow-up work. Its role is to provide the simplest termination mechanism used later in the thesis: one-step invariance of a balanced polytope under a rescaled matrix family.

\section{Overview}

\subsection{Motivating Ideas}
The invariant-polytope algorithm certifies an upper bound for the joint spectral radius by constructing a polytope norm that is adapted to the matrix family.

\subsection{Problem Formulation}
After preprocessing from Section~\ref{sec:introPreprocessing}, we obtain a rescaled family $\tilde{\mathcal{A}}$. From the chosen candidate product and the three-member inequality \eqref{eq:three_member}, we already know $\JSR(\tilde{\mathcal{A}}) \ge 1$. Hence it remains to prove $\JSR(\tilde{\mathcal{A}}) \le 1$.
To simplify notation, we write $\mathcal{A}$ for this rescaled family throughout the remainder of the chapter.

\subsection{Core Idea}
The algorithm tries to construct a balanced polytope $\mathcal{P} \subset \R^d$ that is invariant under $\mathcal{A}$. The associated Minkowski functional then yields a norm with respect to which all matrices in $\mathcal{A}$ have norm at most $1$. This is the basic termination pattern of the thesis. Chapters~\ref{ch:finiteTree} and~\ref{ch:hybridApproach} replace this direct invariance condition by more flexible tree-based bounds.

\section{Algorithmic Formulation}
\label{sec:invAlgorithmStructure}

\subsection{Balanced Polytopes and Polytope Norms}
\begin{definition}
    \label{def:balancedPolytope}
    A polytope $\mathcal{P} \subset \R^d$ is called \emph{balanced} if it spans $\R^d$ and is equal to the symmetrized convex hull of a finite set of vertices. Thus there exists a finite set $V = \{v_1, \ldots, v_M\}$ such that
    \[
    \mathcal{P} = \mathrm{co}_{\mathrm{s}}(V) := \mathrm{co}(V,-V)
    = \left\{ x \in \R^d : x = \sum_{i=1}^{M} \alpha_i v_i \text{ and } \sum_{i=1}^{M} |\alpha_i| \le 1 \right\}.
    \]
    If $\mathcal{A}\mathcal{P} \subseteq \mathcal{P}$ for a family $\mathcal{A}$, then $\mathcal{P}$ is called \emph{invariant under $\mathcal{A}$}.
\end{definition}

\begin{definition}\citep{MagarilIlyaevTikhomirov2001ConvexAnalysis}
    \label{def:minkowskiFunctional}
    Let $C \subseteq \R^d$ be a nonempty convex set with $0 \in \operatorname{int}(C)$. The \emph{Minkowski functional} associated to $C$ is the mapping $p_C : \R^d \to [0,\infty)$ defined by
    \[
    p_C(x) := \inf\left\{ \lambda > 0 \,\middle|\, x \in \lambda C \right\}
    \quad \text{for all } x \in \R^d.
    \]
\end{definition}

\begin{remark}
    \label{rem:balancedPolytopeInducedNorm}
    If $\mathcal{P} = \mathrm{co}_{\mathrm{s}}(V)$ is a balanced polytope, then the Minkowski functional defines a norm on $\R^d$. We denote this norm by
    \[
    \|x\|_{\mathcal{P}} = \|x\|_{\mathrm{co}_{\mathrm{s}}(V)} = \|x\|_V.
    \]
    Moreover,
    \[
    x \in \mathcal{P}
    \quad \Longleftrightarrow \quad
    \|x\|_{\mathcal{P}} \le 1.
    \]
    The induced matrix norm on $\R^{d \times d}$ is submultiplicative as well.
\end{remark}

\subsection{Algorithm Description}
Algorithm~\ref{alg:invariantPolytope} follows the real invariant-polytope construction from \citep[Section~2]{GuglielmiProtasov2013ExactComputationJSR}. Starting from an initial finite set of vertices
\[
V_0 = \{v_1,\ldots,v_M\},
\]
each iteration forms the current balanced polytope
\[
\mathcal{P}_i := \mathrm{co}_{\mathrm{s}}(V_i).
\]
We denote by $R_i \subseteq V_i$ the set of vertices that still have to be tested in iteration $i$.
For every $v \in R_i$ and every matrix $A \in \mathcal{A}$, the algorithm tests whether $Av$ already belongs to $\mathcal{P}_i$.
By Remark~\ref{rem:balancedPolytopeInducedNorm}, this is equivalent to testing whether $\|Av\|_{V_i} \le 1$.
If $Av \notin \mathcal{P}_i$, then $Av$ is added to the next vertex set.
The next test set $R_{i+1}$ consists exactly of the vertices added in iteration $i$.
If $R_i = \emptyset$, then the algorithm terminates and returns $\mathcal{P}_i$.

\subsection{Pseudocode}
\begin{algorithm}[ht]
\caption{Invariant-polytope algorithm}
\label{alg:invariantPolytope}
\begin{algorithmic}
\State \Input{$\mathcal{A}$ (rescaled family of matrices), $V_0$ (set of starting vertices)}
\State $R_0 \gets V_0$
\State $i \gets 0$
\While{$R_i \ne \emptyset$}
\State $V_{i+1} \gets V_i$
\For{$v \in R_i$}
\State $V_{i+1} \xleftarrow{\cup} \{Av : A \in \mathcal{A} \text{ and } \lVert Av \rVert_{\mathrm{co}_{\mathrm{s}}(V_i)} > 1\}$
\EndFor
\State $R_{i+1} \gets V_{i+1} \setminus V_i$
\State $i \gets i + 1$
\EndWhile
\State \Return $\mathrm{co}_{\mathrm{s}}(V_i)$
\end{algorithmic}
\end{algorithm}

\section{Termination Theory}
\label{sec:invExtremalNorms}

\subsection{Proof Strategy}
The invariant-polytope argument is the simplest termination argument in the thesis. First, one shows that an invariant balanced polytope yields an extremal norm. Second, one proves that algorithm termination implies exactly this invariance, because the algorithm checks all one-step images of the current vertices. Later chapters keep the same overall logic but replace one-step invariance by tree-based decompositions and polynomial bounds.

\subsection{Auxiliary Results}
\begin{definition}\citep{Protasov1996JSRInvariantSets}
    \label{def:extremalNorm}
    A norm $\lVert \cdot \rVert$ on $\R^d$ is called \emph{extremal} for a family of matrices $\mathcal{A}$ if
    \[
    \max_{A \in \mathcal{A}} \lVert A x \rVert \leq \JSR(\mathcal{A}) \lVert x \rVert
    \quad \text{for all } x \in \R^d.
    \]
\end{definition}

\begin{theorem}
    \label{thm:extremalNormProductGrowth}
    For every extremal norm $\lVert \cdot \rVert$ of a family of matrices $\mathcal{A}$,
    \[
    \max_{P \in \mathcal{A}^k} \lVert P \rVert = \JSR(\mathcal{A})^k
    \]
    holds for every $k \in \N$.
\end{theorem}
\begin{proof}
    We use the same symbol $\lVert \cdot \rVert$ for the vector norm and its induced matrix norm.
    Let $P = A_{i_k} \cdots A_{i_1} \in \mathcal{A}^k$. By submultiplicativity and Definition~\ref{def:extremalNorm},
    \[
    \lVert P \rVert
    \le \prod_{j=1}^{k} \lVert A_{i_j} \rVert
    \le \JSR(\mathcal{A})^k.
    \]
    Taking the maximum over all products of length $k$ gives
    \[
    \max_{P \in \mathcal{A}^k} \lVert P \rVert \le \JSR(\mathcal{A})^k.
    \]
    The reverse inequality is the right-hand side of the three-member inequality \eqref{eq:three_member}, applied to the extremal norm:
    \[
    \JSR(\mathcal{A}) \le \max_{P \in \mathcal{A}^k} \lVert P \rVert^{1/k}.
    \]
    Raising this estimate to the $k$-th power yields the opposite inequality.
\end{proof}

\begin{remark}
    \label{rem:extremalNormExactComputation}
    In particular, Theorem~\ref{thm:extremalNormProductGrowth} with $k=1$ yields
    \[
    \max_{A \in \mathcal{A}} \lVert A \rVert = \JSR(\mathcal{A}).
    \]
    Thus an extremal norm makes the joint spectral radius directly visible at the level of single matrices.
\end{remark}

\subsection{Main Result}
\begin{theorem}\citep[cf. Section~2]{GuglielmiProtasov2013ExactComputationJSR}
    \label{thm:invariantPolytopeImpliesJsrUpperBound}
    If a balanced polytope $\mathcal{P}$ is invariant under the rescaled family $\mathcal{A}$ from Section~\ref{sec:introPreprocessing}, then the Minkowski functional $\lVert \cdot \rVert_{\mathcal{P}}$ is an extremal norm for $\mathcal{A}$ and
    \[
    \JSR(\mathcal{A}) \le 1.
    \]
\end{theorem}

\begin{proof}
    Let $A \in \mathcal{A}$ and let $x \in \R^d$ satisfy $\|x\|_{\mathcal{P}} \le 1$. By Remark~\ref{rem:balancedPolytopeInducedNorm}, this means $x \in \mathcal{P}$. Since $\mathcal{P}$ is invariant under $\mathcal{A}$, we obtain $Ax \in \mathcal{P}$ and therefore
    \[
    \|Ax\|_{\mathcal{P}} \le 1.
    \]
    By homogeneity, this implies
    \[
    \|A\|_{\mathcal{P}} \le 1
    \quad \text{for all } A \in \mathcal{A}.
    \]
    Because the family has been rescaled so that $\JSR(\mathcal{A}) \ge 1$, Definition~\ref{def:extremalNorm} shows that $\|\cdot\|_{\mathcal{P}}$ is an extremal norm for $\mathcal{A}$. Applying Theorem~\ref{thm:extremalNormProductGrowth} with $k=1$, we obtain
    \[
    \JSR(\mathcal{A}) = \max_{A \in \mathcal{A}} \|A\|_{\mathcal{P}} \le 1.
    \]
\end{proof}

\subsection{Algorithmic Termination}
\begin{theorem}
    \label{thm:invariantPolytopeTermination}
    If Algorithm~\ref{alg:invariantPolytope} terminates and returns $\mathcal{P} = \mathrm{co}_{\mathrm{s}}(V)$, then $\mathcal{P}$ is invariant under $\mathcal{A}$. Consequently,
    \[
    \JSR(\mathcal{A}) \le 1.
    \]
\end{theorem}

\begin{proof}
    At termination, every tested image already belongs to the returned polytope. Hence
    \[
    Av \in \mathcal{P}
    \quad \text{for all } A \in \mathcal{A} \text{ and } v \in V.
    \]
    Now take an arbitrary $x \in \mathcal{P}$. By Definition~\ref{def:balancedPolytope}, there exist coefficients $\alpha_1, \ldots, \alpha_M$ such that
    \[
    x = \sum_{i=1}^{M} \alpha_i v_i
    \quad \text{and} \quad
    \sum_{i=1}^{M} |\alpha_i| \le 1.
    \]
    Therefore, for every $A \in \mathcal{A}$,
    \[
    Ax = \sum_{i=1}^{M} \alpha_i A v_i.
    \]
    Since $\mathcal{P}$ is symmetric and each vector $A v_i$ belongs to $\mathcal{P}$, the vector $\operatorname{sign}(\alpha_i) A v_i$ belongs to $\mathcal{P}$ as well. Because $0 \in \mathcal{P}$ and $\sum_{i=1}^{M} |\alpha_i| \le 1$, convexity yields
    \[
    Ax
    = \sum_{i=1}^{M} |\alpha_i| \operatorname{sign}(\alpha_i) A v_i
      + \left(1-\sum_{i=1}^{M} |\alpha_i|\right) 0
    \in \mathcal{P}.
    \]
    Thus $Ax \in \mathcal{P}$ for every $x \in \mathcal{P}$ and every $A \in \mathcal{A}$, which proves $\mathcal{A}\mathcal{P} \subseteq \mathcal{P}$. The conclusion follows from Theorem~\ref{thm:invariantPolytopeImpliesJsrUpperBound}.
\end{proof}

\begin{remark}
    Theorem~\ref{thm:invariantPolytopeTermination} explains why termination certifies the chosen candidate scaling. A separate question is when finite termination actually occurs. For the real invariant-polytope algorithm, sharp finite-termination criteria are tied to dominant products with unique simple leading eigenvalues; see \citep[Theorem~4]{GuglielmiProtasov2013ExactComputationJSR}. Later refinements extend the method to several competing candidates and better conditioned initial data \citep{GuglielmiProtasov2016InvariantPolytopes,Mejstrik2020ImprovedInvariantPolytopeAlgorithm}.
\end{remark}

\section{Implementation Details}

\subsection{Stopping Criterion and Candidate Rejection}
\label{sec:invStoppingCriterion}
For a candidate product
\[
\Pi = A_{d_k} \cdots A_{d_1},
\]
the real invariant-polytope algorithm is formulated under the assumption that $\Pi$ has a unique simple leading eigenvalue. Let $v_1$ be a corresponding leading eigenvector and let $v_1^*$ be the leading eigenvector of $\Pi^*$, normalized by
\[
(v_1,v_1^*) = 1.
\]
For the cyclic permutations of $\Pi$, one obtains corresponding dual leading eigenvectors $v_j^*$. The stopping criterion from \citep[Section~2.4]{GuglielmiProtasov2013ExactComputationJSR} then says: if during the iteration a tested image $Av$ satisfies
\[
\lVert Av \rVert_{\mathrm{co}_{\mathrm{s}}(V)} > 1
\]
and there exists an index $j$ such that
\[
|(v_j^*,Av)| > 1,
\]
then the chosen candidate cannot be an s.m.p.\ and must be rejected.

\subsection{Rebalancing and Additional Initial Vertices}
\label{sec:invRebalancing}
The basic algorithm is most transparent when there is a single dominant candidate. In practical problems, however, several nearly maximizing products may compete, and the initial polytope may be badly conditioned. Later work therefore introduces balancing of several candidate roots and additional initial vertices in order to improve conditioning and finite-termination behavior \citep{GuglielmiProtasov2016InvariantPolytopes,Mejstrik2020ImprovedInvariantPolytopeAlgorithm}. These refinements are not needed for the basic correctness proof above, but they are important in implementations and form part of the background for the hybrid chapter.

\section{Remarks}

\subsection{Complex Variants}
\label{sec:invEigenvectorCases}
If the leading eigenvector associated with a candidate product is complex, then a real balanced-polytope norm is no longer sufficient and must be replaced by a complex-balanced polytope norm. This changes the convex-hull membership test, while the high-level structure of the algorithm remains the same.

In this thesis, we restrict attention to the real case. For details on complex variants, we refer to \citep{GuglielmiWirthZennaro2005ComplexPolytopeExtremality,GuglielmiZennaro2007BalancedComplexPolytopes,GuglielmiProtasov2013ExactComputationJSR}.

\begin{remark}
    Finite termination of the invariant-polytope algorithm is not guaranteed for every irreducible family. Even when extremal norms exist, they need not be polyhedral, and in that case the algorithm need not terminate with an exact certificate. It may still provide useful bounds, but it need not construct a finite invariant polytope \citep{GuglielmiProtasov2013ExactComputationJSR}.
\end{remark}
