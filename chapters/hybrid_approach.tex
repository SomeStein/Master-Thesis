
\chapter{Invariant-Tree Algorithm}
\label{ch:hybridApproach}

This chapter contains the thesis-specific contribution: a detailed formulation of the invariant-tree algorithm and a corresponding termination analysis, based on the hybrid framework of \citep{MejstrikReif2025HybridApproachJSR}. In the recent literature, the same construction is also referred to as the \emph{tree-flavoured-invariant-polytope-algorithm}. Implementation details for finite $1$-bounding of generated product sets are taken from the cited work and are not claimed as original contributions here.

\section{Overview}
\subsection{Combination Strategies}
\label{sec:hybridStrategies}
At its core, the invariant-polytope algorithm constructs a polytope norm adapted to the matrix family, while the finite-tree algorithm controls growth through product decompositions. This naturally suggests combining both algorithms.

One option is modular: first run the invariant-polytope algorithm for several iterations, then use the resulting norm inside the finite-tree phase. A second option is concurrent: update norm information and tree-decomposition information in each iteration. Following \citep{MejstrikReif2025HybridApproachJSR}, we use the concurrent variant.

\subsection{Problem Formulation}
\label{sec:hybridProblemFormulation}
After preprocessing from Section~\ref{sec:introPreprocessing}, we obtain a scaled family $\tilde{\mathcal{A}}$. From the chosen candidate and the three-member inequality \eqref{eq:three_member}, we already know $\JSR(\tilde{\mathcal{A}}) \geq 1$. Hence proving $\JSR(\tilde{\mathcal{A}}) \leq 1$ is sufficient.

To do this, we show that there exist a polynomial $p$ and a norm $\lVert \cdot \rVert$ such that, similarly to Chapter~\ref{ch:finiteTree}, every product can be bounded polynomially in its length:
\[
\lVert P \rVert \leq p(k) \quad \text{for all } P \in \tilde{\mathcal{A}}^k, \ k \in \N.
\]
The polynomial bound is obtained by decomposing each multi-index into finitely many parts arising from $1$-$v$-bounded test trees, plus one residual part controlled by finite-tree arguments.

\section{Algorithmic Formulation}
\label{sec:hybridAlgorithm}
The hybrid algorithm uses a vertex-wise version of the $1$-boundedness concept from Chapter~\ref{ch:finiteTree}.

\begin{definition}
\label{def:oneVBoundedTree}
Let $v \in \R^d$ and let $\lVert \cdot \rVert$ be a vector norm on $\R^d$.
An $\AG$-tree $T$ is called \emph{$1$-$v$-bounded} with respect to $\lVert \cdot \rVert$ if
\[
\sup_{L \in \mathcal{L}(T)} \lVert Lv \rVert \leq 1.
\]
If
\[
\sup_{L \in \mathcal{L}(T)} \lVert Lv \rVert < 1,
\]
then $T$ is called \emph{strictly $1$-$v$-bounded}.
\end{definition}

The algorithm starts, as in the invariant-polytope method, from a set $V_0$ of starting vertices.
Each vertex set $V_r$ defines a vector polytope norm $\lVert \cdot \rVert _{V_r}$.
At iteration $r$, for every vertex $v \in V_r$, we build a test $\AG$-tree $T_v$.
If some $L$ in the leafage of $T_v$ satisfies $\lVert Lv \rVert _{V_r} \ge 1$, then $Lv$ is added as a new vertex candidate.
If all test-tree leafages in one iteration are $1$-$v$-bounded with respect to the current polytope norm, the algorithm terminates.

\vspace{1cm}

\begin{algorithm}
  \caption{Invariant-tree algorithm}
  \label{alg:hybridApproach}
  \begin{algorithmic}
      \State \Input{$\mathcal{A}$ (set of matrices), $\mathcal{G}$ (set of generators), $V$ (set of starting vertices)}
      \State $V_{\text{new}} \gets V$
      \While {$V_{\text{new}} \ne \emptyset$}
          \State $V_{\text{rem}} \gets V_{\text{new}}$
          \State $V_{\text{new}} \gets \emptyset$
          \For {$v \in V_{\text{rem}}$}
              \State Construct minimal $\AG$-tree $\mathbf{T}$ as discussed in Section~\ref{sec:hybridImplementation}
              \State $\mathcal{W} \gets \{Lv : L \in \mathcal{L}(\mathbf{T}) \text{ and } \lVert Lv \rVert _{\text{co}_{\text{s}}(V)} \geq 1\}$
              \If {$\mathcal{W} \ne \emptyset$}
                  \State $V \xleftarrow{\cup} \mathcal{W}$
                  \State $V_{\text{new}} \xleftarrow{\cup} \mathcal{W}$
              \EndIf  
          \EndFor
      \EndWhile 
  \end{algorithmic}
\end{algorithm}

\section{Termination Theory}
\label{sec:hybridTerminationResults}

\begin{theorem}
  \label{thm:hybridTermination}
  If Algorithm~\ref{alg:hybridApproach} terminates, then $\JSR(\mathcal{A}) \leq 1$.
\end{theorem}

\subsection{Proof Strategy}
\label{subsec:hybridProofOutline}
Products are bounded by norms built during intermediate iterations, and submultiplicativity allows us to pass to the final norm. Therefore, we work with $\lVert \cdot \rVert _V$ for the final vertex set $V = \cup_{i=1}^{K} V_i$, where $K$ is the final iteration.
The proof generalizes the finite-tree argument by repeatedly expanding linear combinations and controlling their coefficients via $1$-$v$-boundedness.

\subsection{Auxiliary Termination Results}

\begin{lemma}
  \label{lem:hybridArbitraryProductBound}
  If Algorithm~\ref{alg:hybridApproach} terminates, there exists a monotone polynomial $p$ such that for every product $P \in \mathcal{A}^k$ and $v \in V$, $\lVert Pv \rVert _{\text{co}_{\text{s}}(V)} \le p(k)$ holds.
\end{lemma}

\begin{proof}
If the algorithm terminates, then the set of vertices $V = \{ v_1, \cdots, v_m \}$ is finite and for each $v \in V$ there exists an $\AG\text{-tree } T_v$ that is $1$-$v$-bounded with respect to $\lVert \cdot \rVert _{\text{co}_{\text{s}}(V)}$. In particular, for every $L \in \mathcal{L}(T_v)$ one has $\lVert Lv \rVert _{\text{co}_{\text{s}}(V)} \leq 1$, which means there exists $\lambda_1, \cdots, \lambda_m$ such that $Lv = \sum \lambda_j v_j$ with $ \sum |\lambda_j| \leq 1 $. According to Lemma~\ref{lem:finitePolynomialBound}, there exists a polynomial $p_v$ such that for every product $P \in \mathcal{A}^k$ encoded by $T_v$, $\lVert Pv \rVert _{\text{co}_{\text{s}}(V)} \leq p_v(k)$.
Take an arbitrary product $P \in \mathcal{A}^k$ and $v \in V$.
Define: 
$$
\begin{aligned}
  p & := \sum \limits_{v \in V} p_v \\
  P' L & := P \text{ with } L  \in  \mathcal{L}(T_v) \text{ if } P \notin T_v \text{ and } L  := I \text{ else}\\
  P'_{i_1 \dots i_r} L_{i_1 \dots i_r} & := P'_{i_1 \dots i_{r-1}} \text{ with } L_{i_1 \dots i_r} \in  \mathcal{L}(T_{v_{i_{r}}}) \text{ if } P'_{i_1 \dots i_{r-1}} \notin T_{v_{i_{r}}}\\
  \text{ and } L_{i_1 \dots i_r} & := I \text{ else.} \\
  \text{Where } i_j \in \mathcal{I} & := \{1, \cdots, m \}\\
\end{aligned}
$$
These decompositions exist and are unique because of the special structure of $\AG\text{-trees}$; details can be found in \citep{MoellerReif2014TreeBasedJSR}.
\newline
Now we have:
$$
\begin{aligned}
  \|Pv\|_{co_s(V)} & = \|P'Lv\|_{co_s(V)} \\ 
  & = \|P' \sum \limits_{i_1 \in \mathcal{I}} \lambda_{i_1} v_{i_1} \|_{co_s(V)} \\ 
  & = \sum \limits_{i_1 \in \mathcal{I}} |\lambda_{i_1}| \|P'v_{i_1} \|_{co_s(V)}\\
  & = \sum \limits_{i_1 \in \mathcal{I}} |\lambda_{i_1}| \|P'_{i_1}L_{i_1}v_{i_1} \|_{co_s(V)} \\ 
  & = \sum \limits_{i \in \mathcal{I}^2} |\lambda_{i_1}| |\lambda_{i_2}| \|P'_{i}v_{i_2} \|_{co_s(V)}\\
  & \cdots \\
  & = \sum \limits_{i \in \mathcal{I}^{k}} |\lambda_{i_1}| \cdot ... \cdot |\lambda_{i_k}| \|P'_{i}v_{i_{k}} \|_{co_s(V)}\\
  & \leq \sum \limits_{i \in \mathcal{I}^{k}} |\lambda_{i_1}| \cdot ... \cdot |\lambda_{i_k}| \text{ } p_{v_{i_k}}(k) \\
  & \leq p(k)\sum \limits_{i \in \mathcal{I}^{k}} |\lambda_{i_1}| \cdot ... \cdot |\lambda_{i_k}| \\
  & =  p(k)\sum \limits_{i_1 \in \mathcal{I}} |\lambda_{i_1}| \sum \limits_{i_2 \in \mathcal{I}} |\lambda_{i_2}| \text{ } ... \sum \limits_{i_k \in \mathcal{I}} |\lambda_{i_k}| \\
  & \leq p(k) \cdot 1 \cdot ... \cdot 1\\
  & \leq p(k)
\end{aligned}
$$
Here the first inequality works since after $k$ steps the prefix $P'_{i}$ is guaranteed to be in the tree $T_{v_{i_k}}$, and thus the norm is bounded by $p_{v_{i_k}}(k)$, which is less than $p(k)$ by definition. The second-to-last inequality follows because the sums of absolute coefficients in the corresponding linear combinations are bounded by $1$ due to $1$-$v$-boundedness of the $\AG\text{-trees}$.
\end{proof}

\subsection{Main Termination Result}

\begin{proof}
  Due to the auxiliary lemmata, Theorem~\ref{thm:hybridTermination} is now straightforward.
  Lemma~\ref{lem:hybridArbitraryProductBound} implies that for every $P \in \mathcal{A}^k$ and $v \in V$, $\lVert Pv \rVert _{\text{co}_{\text{s}}(V)} \le p(k)$.
  In particular, this implies
  $$
    \lVert P \rVert _{\text{co}_{\text{s}}(V)} \le p(k) \quad \text{for all } P \in \mathcal{A}^k.
  $$
  Taking the definition of $\JSR$ gives
  $$
    \JSR(\mathcal{A}) = \lim_{k \to \infty} \max_{P \in \mathcal{A}^k} \lVert P \rVert^\frac{1}{k} \le \lim_{k \to \infty} p(k)^\frac{1}{k} = 1.
  $$ 
\end{proof}

\begin{corollary}
  If the chosen trees are always $T_v = \mathcal{A}$, then Algorithm~\ref{alg:hybridApproach} reduces to the invariant-polytope algorithm~\ref{alg:invariantPolytope}. Hence, the invariant-tree algorithm generalizes the invariant-polytope algorithm.
\end{corollary}

\begin{corollary}
  Assume Algorithm~\ref{alg:hybridApproach} terminates and all test trees $T_v$ are chosen without generator children. Then the polynomial bound in Lemma~\ref{lem:hybridArbitraryProductBound} can be chosen constant, i.e. there exists $C>0$ such that
  \[
  \lVert P \rVert_{\mathrm{co}_{\mathrm{s}}(V)} \le C
  \quad \text{for all } P \in \mathcal{A}^k,\ k\in\N.
  \]
  Consequently, for the rescaled family $\widehat{\mathcal{A}}:=C^{-1}\mathcal{A}$, one has
  \[
  \widehat{\mathcal{A}}\,\mathrm{co}_{\mathrm{s}}(V)\subseteq \mathrm{co}_{\mathrm{s}}(V),
  \]
  so polytope invariance is recovered after scaling.
\end{corollary}

\begin{corollary}
  Assume Algorithm~\ref{alg:hybridApproach} terminates and all test trees coincide, i.e. $T_v=T$ for all $v\in V$. Then the product bound can be derived via a single prefix-leaf decomposition (PL decomposition) in matrix norm, as in Theorem~\ref{thm:finiteTreeTermination}, without vertex-wise recursive decompositions.

  More precisely, if
  \[
  A_I=A_{I_r}\cdots A_{I_1}
  \]
  is the PL decomposition induced by $T$, then
  \[
  \lVert A_I\rVert_{\mathrm{co}_{\mathrm{s}}(V)}
  \le
  \lVert A_{I_r}\rVert_{\mathrm{co}_{\mathrm{s}}(V)}
  \prod_{j=1}^{r-1}\lVert A_{I_j}\rVert_{\mathrm{co}_{\mathrm{s}}(V)}
  \le
  \lVert A_{I_r}\rVert_{\mathrm{co}_{\mathrm{s}}(V)}
  \le p(k),
  \]
  where $p$ is the polynomial from Lemma~\ref{lem:finitePolynomialBound}.
\end{corollary}

This tree depends only on $\mathcal{A}$ and the chosen generator $G$. It is denoted by $T_{\AG}$ and, when $\mathcal{A}$ is clear from context, shortened to $T_\mathcal{G}$.

In principle, the invariant-tree algorithm allows a different test tree for each vertex $v \in V$. More precisely, both the test tree and the corresponding bounding tree can be chosen independently for each vertex, as long as they satisfy the $\AG\text{-tree}$ conditions from Definition~\ref{def:agTree}.

In our implementation, we use minimal trees only. This means that for each vertex we choose a tree that keeps the generator branch and the first covered-node effect, while minimizing the size of leafage to be tested. This reduces the number of norm checks $\lVert Lv \rVert _{\text{co}_{\text{s}}(V)}$ in each iteration and is the default tree-selection policy.

Because generator branches encode families such as $\{G^r : r \in \N\}$, a naive implementation would require infinitely many tests. Following Section~3.3 of \citep{MejstrikReif2025HybridApproachJSR}, we split the generator dynamics into a leading-eigenspace part and a stable remainder, estimate the remainder by pseudo-spectral-radius bounds, and then reduce each infinite branch check to finitely many verifiable conditions for the current vertex.

The practical implementation also uses a maximal iteration counter $r_{\max}$. If no certificate is found before reaching $r_{\max}$, the algorithm stops and returns an approximation interval for $\JSR(\mathcal{A})$ via the three-member inequality:
\[
\underline{\rho}\ \le\ \JSR(\mathcal{A})\ \le\ \overline{\rho}.
\]
In practice, these bounds can already be accurate even when full termination is not achieved.

From a computational perspective, runtime is dominated by norm/membership checks over leafage elements and by tree construction. Memory is dominated by the current vertex set $V$ and temporary tree data. Restricting to minimal trees keeps leafage small, so both runtime and memory growth are significantly lower than with larger test trees.

Finally, candidate handling from the pure invariant-polytope algorithm is still used in the invariant-tree setting. In particular, candidate rejection and candidate updates can be applied before or during invariant-tree iterations to improve robustness.

\section{Remarks}
\label{sec:hybridRemarks}

\subsection{Polytope Invariance}
\label{subsec:hybridPolytopeInvariance}
After Algorithm~\ref{alg:hybridApproach} terminates, the polytope $\text{co}_{\text{s}}(V)$ is, in general, not invariant under the full family $\mathcal{A}$. This is expected, since the invariant-tree algorithm enforces polynomial product bounds rather than one-step invariance under each matrix in $\mathcal{A}$.

The termination certificate is first vertex-wise. For every $v \in V$, the algorithm provides a test tree $T_v$ such that each leaf matrix in $\mathcal{L}(T_v)$ maps $v$ back into the polytope. Equivalently, with
\[
\Gamma_V := \{(v,L) : v \in V,\ L \in \mathcal{L}(T_v)\},
\]
one has
\[
(v,L) \in \Gamma_V
\quad \Longrightarrow \quad
Lv \in \text{co}_{\text{s}}(V).
\]

A stronger, uniform statement is obtained if one can construct a single $\AG$-tree $T_\star$ that bounds all vertices simultaneously, i.e.
\[
\lVert Lv \rVert_{\text{co}_{\text{s}}(V)} \le 1
\quad \text{for all } v \in V \text{ and } L \in \mathcal{L}(T_\star).
\]
Then $T_\star$ induces leafage invariance on the whole polytope:
\[
x \in \text{co}_{\text{s}}(V),\ L \in \mathcal{L}(T_\star)
\quad \Longrightarrow \quad
Lx \in \text{co}_{\text{s}}(V).
\]
Indeed, writing $x=\sum_{i=1}^M \alpha_i v_i$ with $\sum_{i=1}^M |\alpha_i|\le 1$ and $v_i \in V$, one gets
\[
Lx=\sum_{i=1}^M \alpha_i Lv_i,
\]
and each term $Lv_i$ belongs to $\text{co}_{\text{s}}(V)$ by assumption. By convexity and symmetry of $\text{co}_{\text{s}}(V)$, this implies $Lx \in \text{co}_{\text{s}}(V)$.

The existence and efficient construction of such a global tree $T_\star$ from the family $(T_v)_{v\in V}$ is a separate algorithmic problem and remains part of future work.

This invariance interpretation, together with the intended play on words relative to the invariant-polytope algorithm, motivates the name \emph{invariant-tree algorithm}.

\subsection{Memory and Time Complexity}
\label{subsec:hybridComplexity}

We summarize the computational complexity in terms of the final vertex set
\[
V=\{v_1,\dots,v_M\},\qquad M:=|V|.
\]
By Algorithm~\ref{alg:hybridApproach}, each vertex is processed once after it is added to $V$. For a processed vertex $v$, write
\[
\ell_v := |\mathcal{L}(T_v)|
\]
for the size of the tested leafage, and denote by $C_{\mathrm{tree}}(v)$ the cost of constructing $T_v$.

For each leaf matrix $L\in\mathcal{L}(T_v)$, one needs to evaluate $\|Lv\|_{\mathrm{co}_{\mathrm{s}}(V)}$, which consists of
\[
C_{\mathrm{mv}}(d)\ \text{(matrix-vector product)}\quad +\quad C_{\mathrm{norm}}(d,|V|)\ \text{(polytope-norm / membership test)}.
\]
Hence the total runtime is bounded by
\[
T_{\mathrm{total}}
\;\lesssim\;
\sum_{v\in V}\Big(C_{\mathrm{tree}}(v)+\ell_v\big(C_{\mathrm{mv}}(d)+C_{\mathrm{norm}}(d,|V|)\big)\Big).
\]
With uniform bounds $C_{\mathrm{tree}}(v)\le C_{\mathrm{tree}}^{\max}$ and $\ell_v\le \ell_{\max}$, this yields
\[
T_{\mathrm{total}}
=
\mathcal{O}\!\left(
M\Big(C_{\mathrm{tree}}^{\max}+\ell_{\max}\big(C_{\mathrm{mv}}(d)+C_{\mathrm{norm}}(d,M)\big)\Big)
\right).
\]

Generator branches introduce an additional preprocessing/verification cost (Section~3.3 in \citep{MejstrikReif2025HybridApproachJSR}), where infinite families are reduced to finitely many checks via spectral estimates and pseudo-spectral-radius bounds. This overhead is typically amortized across vertices when the same generators are reused.

The peak memory footprint has the structure
\[
S_{\mathrm{total}}
\;\lesssim\;
\underbrace{\mathcal{O}(nd^2)}_{\text{input matrices}}
\;+\;
\underbrace{\mathcal{O}(Md)}_{\text{vertex set}}
\;+\;
\underbrace{\mathcal{O}(S_{\mathrm{tree}}^{\max})}_{\text{active tree data}}
\;+\;
\underbrace{\mathcal{O}(S_{\mathrm{norm}}(d,M))}_{\text{norm/membership solver state}},
\]
where $S_{\mathrm{tree}}^{\max}$ denotes the maximal storage needed for one active test tree.

Practically, minimal trees reduce $\ell_v$ and $S_{\mathrm{tree}}^{\max}$ substantially; if all test trees coincide, tree construction and PL-decomposition data can be reused globally (Corollary above), reducing both runtime and memory constants. Nevertheless, no global polynomial-time bound in $(n,d)$ is expected in full generality, consistent with known hardness results for JSR computation \citep{TsitsiklisBlondel1997HardToComputeJSR,BlondelTsitsiklis2000NPHardMaxAlgebra}.
